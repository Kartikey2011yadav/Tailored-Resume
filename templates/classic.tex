%-----------EDUCATION-----------------
\section{Education}
  \resumeSubHeadingListStart
  \BLOCK{for edu in education}
    \resumeSubheading
      {\VAR{edu.institution}}{\VAR{edu.endDate}}
      {\VAR{edu.studyType} in \VAR{edu.area}}{\VAR{edu.location}}
      \BLOCK{if edu.courses}
      \resumeItemListStart
        \resumeItem{Coursework}{\VAR{edu.courses | join(", ")}}
      \resumeItemListEnd
      \BLOCK{endif}
  \BLOCK{endfor}
  \resumeSubHeadingListEnd


%-----------EXPERIENCE-----------------
\section{Experience}
  \resumeSubHeadingListStart
  
  \BLOCK{for job in work}
    \resumeSubheading
      {\VAR{job.company}}{\VAR{job.startDate} -- \VAR{job.endDate}}
      {\VAR{job.position}}{\VAR{job.location}}
      \resumeItemListStart
        \BLOCK{if job.summary}
        \item \small{\VAR{job.summary}}
        \BLOCK{endif}
        \BLOCK{for highlight in job.highlights}
        \item \small{\VAR{highlight}}
        \BLOCK{endfor}
      \resumeItemListEnd
  \BLOCK{endfor}

  \resumeSubHeadingListEnd
